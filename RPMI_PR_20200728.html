How stable are ODs of blank wells, to use for normalisation?

ggplot(data=filter(annotated,Strain==""), 
       aes(x=Time, y=OD595, color=Strain, group = Well)) +

These are very stable, except H2. Remove from normalisation.

Calculate median OD for blank wells for each growth medium

Remove well H2

blank_OD_summary <- annotated %>%
  filter(Strain=="") %>%
  filter(Well != "H2") %>%
  # filter out columns 1 and 12 as well?
  group_by(Medium) %>%
  summarise(OD_median=median(OD595),
            OD_mean=mean(OD595),
            OD_max=max(OD595),
            OD_min=min(OD595))
print(blank_OD_summary)
## # A tibble: 1 x 5

Subtract blank OD to make corrected OD and Plot OD_corrected v Time (hrs)

normalisedOD <- annotated %>%
  left_join(blank_OD_summary, by="Medium") %>%
  mutate(OD_corrected = OD595 - OD_median)

Plot OD all

ggplot(data=normalisedOD %>%
                      filter(Strain != ""), 
                      aes(x=Time, y=OD_corrected, color=Strain, group = Well)) + 
       geom_line() + 
       scale_y_continuous(limits=c(0,0.1),expand=c(0,0))+
       geom_vline(aes(xintercept=15), linetype="dotted", colour="black") +

OD_summary <- normalisedOD %>%
  filter(Strain!="") %>%
  group_by(Time,Strain) %>%
  summarise(median_OD_corrected=median(OD_corrected))
